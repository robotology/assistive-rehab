286  bool getPoint3D(const int u, const int v, yarp::sig::Vector &p) const
287  {
288  if ((u>=0) && (u<depth.width()) && (v>=0) && (v<depth.height()))
289  {
290  double f=depth.width()/(2.0*tan(fov_h*(M_PI/180.0)/2.0));
291  double d=depth(u,v);
292  if ((d>0.0) && (f>0.0))
293  {
294  double x=u-0.5*(depth.width()-1);
295  double y=v-0.5*(depth.height()-1);
296 
297  p=d*ones(3);
298  p[0]*=x/f;
299  p[1]*=y/f;
300 
301  return true;
302  }
303  }
304  return false;
305  }
