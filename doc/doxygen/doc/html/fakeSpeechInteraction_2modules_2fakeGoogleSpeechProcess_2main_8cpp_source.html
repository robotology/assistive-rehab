156  /********************************************************/
157  bool configure(yarp::os::ResourceFinder &rf)
158  {
159  this->rf=&rf;
160  yarp::os::Bottle &general = rf.findGroup("general");
161  if (general.isNull())
162  {
163  yError()<<"Unable to find \"general\"";
164  return false;
165  }
166  std::string moduleName = general.check("name", yarp::os::Value("yarp-google-speech-process"), "module name (string)").asString();
167  int numKey = general.find("num-keywords").asInt32();
168  for (int i=0; i<numKey; i++)
169  {
170  std::string keywi = "keyword-"+std::to_string(i);
171  yarp::os::Bottle &bKeyword = rf.findGroup(keywi);
172  if (bKeyword.isNull())
173  {
174  yError()<<"Unable to find"<<keywi;
175  return false;
176  }
177  if (!bKeyword.check("key") || !bKeyword.check("value"))
178  {
179  yError()<<"Unable to find \"key\" and/or \"value\"";
180  return false;
181  }
182  std::string key = bKeyword.find("key").asString();
183  yarp::os::Bottle *bValue = bKeyword.find("value").asList();
184  std::vector<std::string> value(bValue->size());
185  for (int j=0; j<bValue->size(); j++)
186  {
187  value[j] = bValue->get(j).asString();
188  }
189 
190  key_map[key] = value;
191  }
192 
193  setName(moduleName.c_str());
194 
195  rpcPort.open(("/"+getName("/rpc")).c_str());
196 
197  processing = new FakeProcessing( moduleName, key_map );
198 
199  /* now start the thread to do the work */
200  processing->open();
201 
202  attach(rpcPort);
203 
204  return true;
205  }
