156 /********************************************************/
157 void Rom_Processor::estimate()
158 {
159  Vector v1(3,0.0);
160  Vector ref_dir(3,0.0);
161 
162  //get reference keypoint from skeleton
163  curr_skeleton.normalize();
164  string tag_joint=rom->getTagJoint();
165  if(curr_skeleton[tag_joint]->isUpdated() && curr_skeleton[tag_joint]->getChild(0)->isUpdated())
166  {
167  double theta;
168  if(curr_skeleton[tag_joint]->getNumChild())
169  {
170  if(rom->getTagPlane()=="coronal")
171  {
172  plane_normal=curr_skeleton.getCoronal();
173  }
174  else if(rom->getTagPlane()=="sagittal")
175  {
176  plane_normal=curr_skeleton.getSagittal();
177  }
178  else if(rom->getTagPlane()=="transverse")
179  {
180  plane_normal=curr_skeleton.getTransverse();
181  }
182  Vector k1=toCurrFrame(tag_joint);
183  Vector k2=toCurrFrame(curr_skeleton[tag_joint]->getChild(0)->getTag());
184  v1=k2-k1;
185  v1=projectOnPlane(v1,plane_normal);
186  double n1=norm(v1);
187  if(n1>0.0)
188  v1/=n1;
189  if(!rom->getRefJoint().empty())
190  {
191  Vector k_dir=toCurrFrame(rom->getRefJoint());
192  ref_dir=k1-k_dir;
193  }
194  else
195  {
196  ref_dir=rom->getRefDir();
197  }
198  double n2=norm(ref_dir);
199  double dot_p=dot(v1,ref_dir);
200  theta=acos(dot_p/n2);
201  range=theta*(180/M_PI);
202  prev_range=range;
203  }
204  else
205  {
206  yError() << "The keypoint does not have a child ";
207  range=0.0;
208  }
209  }
210  else
211  {
212  range=prev_range;
213  }
214 }

310 /********************************************************/
311 void Step_Processor::estimateSpatialParams(const Vector& dist)
312 {
313  Vector x({dist[0]}), y({dist[1]});
314 
315  x = filter_dist->filt(x);
316  y = filter_dist->filt(y);
317 
318  Vector difference_filtered(cat(x, y));
319 
320  // Distance between feet as norm to keep invariance
321  // wrt to skeleton orientation
322  Vector feet_euclidean_distance({norm(difference_filtered)});
323  stepdist = feet_euclidean_distance[0];
324 
325  // Use with care: projection on skeleton planes needs precise
326  // keypoints estimation
327  if (enable_plane_projection)
328  {
329  Vector step_sag = projectOnPlane(difference_filtered, curr_skeleton.getSagittal());
330  Vector step_cor = projectOnPlane(difference_filtered, curr_skeleton.getCoronal());
331 
332  steplen = abs(step_sag[0]);
333  stepwidth = abs(step_cor[0]);
334  }
335  else
336  {
337  // Distance along X between keypoints in skeleton frame
338  // Low-accuracy metric to measure step length
339  steplen = abs(difference_filtered[0]);
340 
341  // Distance along Y between keypoints in skeleton frame
342  // Low-accuracy metric to measure step width
343  stepwidth = abs(difference_filtered[1]);
344  }
345 
346  feetdist.push_back(stepdist);
347  tdist.push_back(Time::now());
348 
349  pair<deque<double>,deque<int>> step_params;
350 
351  double tlast=0.0;
352  step_params=findPeaks(feetdist, step_thresh);
353  deque<double> stepvec=step_params.first;
354  deque<int> strikes=step_params.second;
355 
356  if (strikes.size()>0)
357  {
358  tlast=tdist[strikes.back()];
359  }
360 
361  if ((Time::now() - tlast) <= time_window)
362  {
363  int laststrike = stepvec.size()>numsteps ? strikes.back() : 0;
364  numsteps = (int)strikes.size();
365  }
366 }

380 /********************************************************/
381 pair<deque<double>,deque<int>> Step_Processor::findPeaks(const Vector &d, const double &minv)
382 {
383  deque<double> val;
384  deque<int> idx;
385  for(int i=0; i<d.size(); i++)
386  {
387  if (d[i]<=minv)
388  {
389  continue;
390  }
391  if(i==0 && d[i]>=d[i+1])
392  {
393  val.push_back(d[i]);
394  idx.push_back(i);
395  }
396  else if(i==d.size() && d[i]>=d[i-1])
397  {
398  val.push_back(d[i]);
399  idx.push_back(i);
400  }
401  else if(d[i]>=d[i-1] && d[i]>=d[i+1])
402  {
403  val.push_back(d[i]);
404  idx.push_back(i);
405  }
406  }
407  if (val.size()>step_window)
408  {
409  val.pop_front();
410  }
411  pair<deque<double>,deque<int>> step_params;
412  step_params=make_pair(val,idx);
413  return step_params;
414 }

467 /********************************************************/
468 void EndPoint_Processor::estimate()
469 {
470  if(ep->getTagPlane() == "coronal")
471  plane_normal[0]=1.0;
472  else if(ep->getTagPlane() == "sagittal")
473  plane_normal[1]=1.0;
474  else if(ep->getTagPlane() == "transverse")
475  plane_normal[2]=1.0;
476 
477  curr_skeleton.normalize();
478  string tag_joint = ep->getTagJoint();
479  if(curr_skeleton[tag_joint]->isUpdated())
480  {
481  //we compute ideal and current trajectory wrt left/right shoulder
482  Vector ref = first_skeleton[tag_joint]->getParent(0)->getParent(0)->getPoint();
483  ref.push_back(1.0);
484  Vector v = curr_skeleton[tag_joint]->getPoint();
485  v.push_back(1.0);
486  Vector transformed_v = curr_frame*v;
487  Vector transformed_ref = curr_frame*ref;
488  Vector dv = transformed_v.subVector(0,2)-transformed_ref.subVector(0,2);
489  est_traj = getTrajectory(dv);
490 
491  yDebug() << first_skeleton[tag_joint]->getParent(0)->getParent(0)->getTag() << ref.toString();
492  yDebug() << tag_joint << v.toString();
493  cout << endl;
494 
495  Vector t=ep->getTarget();
496  t.push_back(1.0);
497  Vector transformed_t = curr_frame*t;
498  Vector dt = transformed_t.subVector(0,2)-transformed_ref.subVector(0,2);
499  ideal_traj = getTrajectory(dt);
500 
501  //we compute velocity and smoothness of the end-point
502  AWPolyElement el(transformed_v,Time::now());
503  vel=norm(linEst->estimate(el));
504  smoothness = norm(jerkEst->estimate(el));
505 
506  prev_est_traj=est_traj;
507  prev_ideal_traj=ideal_traj;
508  prev_vel=vel;
509  prev_smoothness=smoothness;
510  }
511  else
512  {
513  est_traj=prev_est_traj;
514  ideal_traj=prev_ideal_traj;
515  vel=prev_vel;
516  smoothness=prev_smoothness;
517  }
518 
519 }
