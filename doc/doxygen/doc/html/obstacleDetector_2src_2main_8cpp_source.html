374  double getClosestCluster(const Matrix &c, const vector<int> &ids)
375  {
376  int n=ids.size();

379  Vector dist(n);

393  /****************************************************************/
394  double getClusterDist(const Matrix &c, const int &id)
395  {
396  cv::Point2d center{0.0,0.0};
397  int n=0;
398  for (int i=0;i<c.rows();i++)
399  {
400  if ((int)c[i][2]==id)
401  {
402  center.x+=c[i][0];
403  center.y+=c[i][1];
404  n++;
405  }
406  }
407  center.x/=n;
408  center.y/=n;
409  return sqrt((center.x*center.x)+(center.y*center.y));
410  }

434  /***************************************************************/
435  int removeOutliers(Matrix &c, vector<int> &ids)
436  {
437  Vector id=c.getCol(2);
438  int maxid=yarp::math::findMax(id);
439  int nclusters=0;
440  for (int i=0; i<=maxid; i++)
441  {
442  int npoints=std::count(id.begin(),id.end(),i);
443  if (npoints<=min_points)
444  {
445  update(c,i);
446  }
447  else
448  {
449  ids.push_back(i);
450  nclusters++;
451  }
452  }
453  return nclusters;
