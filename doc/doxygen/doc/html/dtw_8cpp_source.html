47 double Dtw::computeDistance(const vector<double> &s, const vector<double> &t, Matrix &distMat) const
48 {
49  //compute distance matrix
50  int ns=(int)s.size();
51  int nt=(int)t.size();
52  int j1,j2;
53  double cost,temp;
54  for(int i=1;i<=ns;i++)
55  {
56  if(win==-1)
57  {
58  j1=1;
59  j2=nt;
60  }
61  else
62  {
63  j1= i-win>1 ? i-win : 1;
64  j2= i+win<nt ? i+win : nt;
65  }
66  for(int j=j1;j<=j2;j++)
67  {
68  cost=sqrt((s[i-1]-t[j-1])*(s[i-1]-t[j-1]));
69  temp=distMat[i-1][j];
70  if(distMat[i][j-1]!=-1)
71  {
72  if(temp==-1 || distMat[i][j-1]<temp)
73  temp=distMat[i][j-1];
74  }
75  if(distMat[i-1][j-1]!=-1)
76  {
77  if(temp==-1 || distMat[i-1][j-1]<temp)
78  temp=distMat[i-1][j-1];
79  }
80  distMat[i][j]=cost+temp;
81  }
82  }
83 
84  return(distMat[ns][nt]/nt);
85 }
