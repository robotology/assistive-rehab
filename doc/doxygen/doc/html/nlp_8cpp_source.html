39  /****************************************************************/
40  Vector get2D(const Vector &x) const
41  {
42  Vector p(2,0.0);
43  if (x[2]>0.0)
44  {
45  p[0]=camParams.get_focal()*(x[0]/x[2])+(camParams.get_width()-1)/2.0;

51  /****************************************************************/
52  Vector get3D(const Vector &p, const double d) const
53  {
54  Vector x(3,d);
55  x[0]*=(p[0]-(camParams.get_width()-1)/2.0)/camParams.get_focal();
56  x[1]*=(p[1]-(camParams.get_height()-1)/2.0)/camParams.get_focal();

60  /****************************************************************/
61  void compute_quantities(const Ipopt::Number *x)
62  {
63  size_t i=0;
64  for (auto c1=k; c1!=nullptr; c1=c1->getChild(0))
65  {
66  if (auto c2=c1->getChild(0))
67  {
68  p1[i]=get2D(c1->getPoint());
69  y1[i]=get3D(p1[i],x[i]);
70  Dy1[i]=get3D(p1[i],1.0);
71 
72  p2[i]=get2D(c2->getPoint());
73  y2[i]=get3D(p2[i],x[i+1]);
74  Dy2[i]=get3D(p2[i],1.0);
75 
76  d1[i]=y1[i]-y2[i];
77  d2[i]=dot(d1[i],d1[i])-lengths[i]*lengths[i];
78  i++;
79  }
80  else
81  {
82  break;
83  }
84  }
85  }
